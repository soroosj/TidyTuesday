3. Transform data

   library (glue)
   
   #https://en.wikipedia.org/wiki/List_of_spaceflight-related_accidents_and_incidents
   astronaut_fatality <- astronaut_raw %>%
      filter (mission_title %in% c("Soyuz 1","Soyuz 11","STS-107", "STS-51L","STS-51-L", "STS 51-L")) %>%
      pull (mission_title)

   astronaut <- astronaut_raw %>%
      group_by (name) %>%
      arrange (-mission_number) %>%
      filter (mission_title %in% astronaut_fatality) %>%
      slice(1) %>%
      mutate (
         birth_death = glue ("{year_of_birth} - {year_of_mission}"),
         nationality = ifelse (nationality == "U.S.S.R/Russia", "Russia",nationality)
         ) %>%
      rename (
         global_number = number,
         missions_count = total_number_of_missions,
         missions_hours = total_hrs_sum
         ) %>%
      select (name, nationality, birth_death, global_number, nationwide_number, missions_count, missions_hours)
