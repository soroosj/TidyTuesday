3. Transform data

   library (glue)
   library (countrycode)
   
   #https://en.wikipedia.org/wiki/List_of_spaceflight-related_accidents_and_incidents
   astronaut_fatality <- astronaut_raw %>%
      filter (mission_title %in% c("Soyuz 1","Soyuz 11","STS-107", "STS-51L","STS-51-L", "STS 51-L")) %>%
      pull (name)

   astronaut <- astronaut_raw %>%
      group_by (name) %>%
      arrange (-mission_number) %>%
      filter (name %in% astronaut_fatality)%>%
      slice(1) %>%
      mutate (
         name_life = glue ("{name}\n({year_of_birth} - {year_of_mission})"),
         nationality_name = ifelse (nationality == "U.S.S.R/Russia", "Russia",nationality),
         nationality_code = countrycode(nationality_name, origin = "country.name", destination = "iso2c"),
         number_global_nation = glue ("{number} / {nationwide_number}"),
         mission_hours = round(total_hrs_sum,0),
         mission_hours_count= glue ("{mission_hours} / {total_number_of_missions})")
         ) %>%
      left_join (astronaut_photos) %>%
      select (name_life, nationality_code, number_global_nation, mission_title, occupation, mission_hours_count, photo) %>%
      head(10)

4. Visualize - indvidual charts

   library (ggimage)

   #https://github.com/GuangchuangYu/ggimage/issues/6

   ggplot (astronaut) +
      geom_flag(
         aes (x = 1.25, y = 1, image = nationality_code),
         size = 0.25, hjust = 1, vjust = 0
         ) +
      geom_text(
         aes (x = 0, y = 0, label = number_global_nation),
         size = 3, hjust = 0, color = "white"
         ) +
      geom_text(
         aes (x = 1, y = 0, label = mission_hours_count),
         size = 3, hjust = 0, color = "white"
